步骤1：数据库设计(我会提供云数据库，可以跳过步骤1)

sql
-- 用户表
CREATE TABLE users (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    email VARCHAR(100),
    created_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_time DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- 行程表
CREATE TABLE travel_plans (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL,
    destination VARCHAR(200) NOT NULL,
    start_date DATE,
    end_date DATE,
    total_budget DECIMAL(10,2),
    people_count INT,
    preferences TEXT,
    generated_itinerary TEXT, -- AI生成的完整行程
    created_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

-- 费用记录表
CREATE TABLE expenses (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    travel_plan_id BIGINT NOT NULL,
    category VARCHAR(50), -- 交通、住宿、餐饮等
    item_name VARCHAR(200),
    amount DECIMAL(10,2),
    note TEXT,
    created_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (travel_plan_id) REFERENCES travel_plans(id)
);
步骤2：RESTful API设计

text
用户相关：
POST /api/auth/register     # 用户注册
POST /api/auth/login        # 用户登录
POST /api/auth/logout       # 用户退出

行程相关：
GET  /api/travel-plans      # 获取用户所有行程
POST /api/travel-plans      # 创建新行程
GET  /api/travel-plans/{id} # 获取行程详情
PUT  /api/travel-plans/{id} # 更新行程
DELETE /api/travel-plans/{id} # 删除行程

工具接口：
POST /api/voice/transcribe  # 语音转文字
POST /api/ai/generate-plan  # AI生成行程
